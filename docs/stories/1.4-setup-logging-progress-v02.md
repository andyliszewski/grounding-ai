### Step 1: Create `pdf2llm/logging_setup.py`

```python
"""Logging configuration for pdf2llm."""
import logging
import sys
from typing import Optional

def setup_logging(verbose: bool = False) -> logging.Logger:
    """
    Configure logging for pdf2llm.

    Args:
        verbose: If True, set log level to DEBUG. Otherwise INFO.

    Returns:
        Configured logger instance
    """
    logger = logging.getLogger("pdf2llm")

    # Set log level
    logger.setLevel(logging.DEBUG if verbose else logging.INFO)

    # Remove existing handlers to avoid duplicates
    logger.handlers.clear()

    # Console handler (stderr)
    console_handler = logging.StreamHandler(sys.stderr)
    console_handler.setLevel(logging.DEBUG if verbose else logging.INFO)

    # Log format: [timestamp] [level] message
    formatter = logging.Formatter(
        "[%(asctime)s] [%(levelname)s] %(message)s",
        datefmt="%Y-%m-%d %H:%M:%S"
    )
    console_handler.setFormatter(formatter)

    logger.addHandler(console_handler)

    # Don't propagate to root logger
    logger.propagate = False

    return logger


def get_logger(name: Optional[str] = None) -> logging.Logger:
    """
    Get logger instance.

    Args:
        name: Optional logger name (default: "pdf2llm")

    Returns:
        Logger instance
    """
    return logging.getLogger(name or "pdf2llm")
```

### Step 2: Create `pdf2llm/stats.py`

```python
"""Statistics tracker for pdf2llm processing."""
from dataclasses import dataclass, field
from time import time
from typing import List, Dict

@dataclass
class ProcessingStats:
    """Track processing statistics."""

    # Counts
    total_files: int = 0
    processed: int = 0
    succeeded: int = 0
    failed: int = 0
    total_chunks: int = 0

    # Failed files with reasons
    failed_files: List[Dict[str, str]] = field(default_factory=list)

    # Timing
    start_time: float = field(default_factory=time)
    end_time: float = 0.0

    def record_success(self, file_name: str, chunk_count: int) -> None:
        """Record successful file processing."""
        self.processed += 1
        self.succeeded += 1
        self.total_chunks += chunk_count

    def record_failure(self, file_name: str, reason: str) -> None:
        """Record failed file processing."""
        self.processed += 1
        self.failed += 1
        self.failed_files.append({"file": file_name, "reason": reason})

    def finish(self) -> None:
        """Mark processing as finished."""
        self.end_time = time()

    @property
    def duration(self) -> float:
        """Get processing duration in seconds."""
        end = self.end_time if self.end_time > 0 else time()
        return end - self.start_time

    def get_summary(self) -> str:
        """
        Generate summary string.

        Returns:
            Formatted summary text
        """
        lines = [
            "Summary:",
            f"  Files processed: {self.processed}",
            f"  Succeeded: {self.succeeded}",
            f"  Failed: {self.failed}",
        ]

        # List failed files
        if self.failed_files:
            for fail in self.failed_files:
                lines.append(f"    - {fail['file']}: {fail['reason']}")

        lines.append(f"  Total chunks: {self.total_chunks}")
        lines.append(f"  Duration: {self.duration:.1f}s")

        return "\n".join(lines)
```

## Technical Notes

### Logging Best Practices

**Why stderr for logs?**
- Stdout is for program output (summary, results)
- Stderr is for logging and diagnostics
- Allows users to redirect output: `pdf2llm ... > results.txt 2> logs.txt`

**Log Levels:**
- **DEBUG**: Internal state, function calls, detailed operations
- **INFO**: Major milestones (started processing, completed file)
- **WARNING**: Non-fatal issues (skipped file, fallback used)
- **ERROR**: Fatal issues for individual files

### Progress Bar with tqdm

**Why tqdm?**
- Industry standard for Python progress bars
- Auto-detects terminal width
- Shows ETA, speed, percentage
- Compatible with logging (doesn't corrupt output)

**Configuration:**
- `leave=True`: Keep progress bar after completion
- `unit="file"`: Show "files" instead of "it" (iterations)
- `desc`: Description shown before progress bar
- `set_description()`: Update description dynamically

### Statistics Tracking

**Why separate stats class?**
- Centralized state management
- Easy to test
- Reusable in different contexts (CLI, API, tests)
- Clear separation of concerns

**Tracked Metrics:**
- Counts (files, chunks, successes, failures)
- Failed files with reasons (for debugging)
- Timing (start, end, duration)
