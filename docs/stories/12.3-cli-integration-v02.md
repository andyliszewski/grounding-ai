## Story

**As a** user of the local agent system,
**I want** to enable agentic mode via a CLI flag,
**So that** I can choose between simple RAG and autonomous tool-calling behavior.

3. Verbose mode (`-v`) shows tool calls when in agentic mode

5. Help text updated with `--agentic` option and description
6. `--max-iterations` flag allows configuring iteration limit
7. Error messages clearly indicate which mode is active
8. Both modes work with the same `--agent`, `--query`, and other flags

### Agentic Query Function

```python
from agentic import run_agentic_loop, AgenticConfig, ToolRegistry
from search_corpus_tool import create_search_corpus_tool

def run_agentic_query(args, index, chunk_map, embedder, persona):
    """Run a single query in agentic mode."""

    # Create tool registry
    registry = ToolRegistry()
    schema, executor = create_search_corpus_tool(
        index, chunk_map, embedder, Path(args.corpus).expanduser()
    )
    registry.register("search_corpus", schema, executor)

    # Build system prompt from persona
    system_prompt = build_system_prompt(persona, args.agent)

    # Configure agentic loop
    config = AgenticConfig(
        max_iterations=args.max_iterations,
        verbose=args.verbose
    )

    # Run the loop
    result = run_agentic_loop(
        user_query=args.query,
        tools=registry.schemas,
        tool_executor=registry.execute,
        model=detect_model(args.api_url),  # Detect model from API
        api_url=args.api_url,
        config=config,
        system_prompt=system_prompt
    )

    # Format output
    output = result.content

    if args.verbose:
        output = format_verbose_output(result) + "\n\n" + output

    return output


def format_verbose_output(result):
    """Format verbose information about the agentic run."""
    lines = [
        f"[Agentic] Iterations: {result.iterations}",
        f"[Agentic] Stopped: {result.stopped_reason}",
        f"[Agentic] Tool calls made: {len(result.tool_calls_made)}"
    ]

    for tc in result.tool_calls_made:
        lines.append(f"  - {tc.name}({tc.arguments})")

    return "\n".join(lines)
```

### Updated REPL

```python
def run_repl(args, index, chunk_map, embedder, persona):
    """Interactive REPL supporting both modes."""

    # Setup for agentic mode if enabled
    if args.agentic:
        registry = ToolRegistry()
        schema, executor = create_search_corpus_tool(
            index, chunk_map, embedder, Path(args.corpus).expanduser()
        )
        registry.register("search_corpus", schema, executor)

        config = AgenticConfig(
            max_iterations=args.max_iterations,
            verbose=args.verbose
        )

    system_prompt = build_system_prompt(persona, args.agent)
    show_sources = not args.no_sources

    # Display greeting
    print(persona.get("greeting", f"Agent {args.agent} ready."))

    # Prompt indicator
    prompt = "[Agentic] You: " if args.agentic else "You: "

    while True:
        try:
            user_input = input(prompt).strip()
        except (EOFError, KeyboardInterrupt):
            print("\nGoodbye!")
            break

        if not user_input:
            continue

        # Handle commands
        if user_input.lower() in ("quit", "q", "exit"):
            print("Goodbye!")
            break

        if user_input.lower() == "sources":
            show_sources = not show_sources
            print(f"Sources: {'ON' if show_sources else 'OFF'}")
            continue

        # Process query
        if args.agentic:
            result = run_agentic_loop(
                user_query=user_input,
                tools=registry.schemas,
                tool_executor=registry.execute,
                model=detect_model(args.api_url),
                api_url=args.api_url,
                config=config,
                system_prompt=system_prompt
            )

            if args.verbose:
                print(format_verbose_output(result))
                print()

            print(result.content)

        else:
            # Existing simple RAG logic
            context = search_corpus(user_input, index, chunk_map, embedder, args.top_k)
            response = query_llm(user_input, context, system_prompt, args.api_url)
            print(response)

            if show_sources:
                print_sources(context)

        print()  # Blank line between exchanges
```
