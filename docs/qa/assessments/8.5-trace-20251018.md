@pytest.mark.formula
def test_extract_formulas_from_pdf():
    """Test complete formula extraction workflow."""
    # Given: PDF with known formulas
    pdf_path = "test_pdfs/simple_equations.pdf"

    # When: Process with formula extraction
    result = subprocess.run([
        sys.executable, "-m", "pdf2llm.cli",
        pdf_path, "/tmp/test_output",
        "--extract-formulas",
        "--formula-format", "both"
    ])

    # Then: Formula files created
    assert (Path("/tmp/test_output") / "simple-equations" / "formulas" / "formula_0001_0001.tex").exists()
    assert (Path("/tmp/test_output") / "simple-equations" / "formulas" / "formula_0001_0001.mathml").exists()

    # And: Manifest includes formula metadata
    manifest = ujson.load(open("/tmp/test_output/_index.json"))
    assert manifest["docs"][0]["formula_metadata"]["formula_count"] > 0
